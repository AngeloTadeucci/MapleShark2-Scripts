using (ScriptAPI) {
    f = AddByte("function");

    switch ((int)f) {
        case 0:
            // opening ui
        case 1: // add item to ui
            AddBool("0 - lock|1 - unlock");
            AddLong("ItemUid");
        case 2: // remove item from ui
            AddLong("ItemUid");
        case 3:
            AddBool("0 - lock|1 - unlock");
    }
}
