    f = AddByte("Function");
    if (f == 8) {
        // SendJob load
    } else if (f == 9) {
        count = AddInt("Count");
        for (i = 0; i < count; i++) {
            StartNode("Skill " + i);
            AddInt("SkillId");
            AddShort("Skill Points");
            AddByte("Unknown");
            EndNode(false);
        }
    }
}
