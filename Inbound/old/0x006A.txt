using (ScriptAPI) {
    AddInt("PlayerObjectId");
    b = AddBool("IsStart");
    if (b) {
        DecodeCubeItemInfo();
        AddInt("Unknown");
    }
}

function DecodeCubeItemInfo() {
    using (ScriptAPI) {
        StartNode("CubeItemInfo");
        AddInt("ItemId");
        AddLong("ItemUid");
        AddLong("Unknown");
        b = AddBool("IsUgc");
        if (b) {
            DecodeUgcData();
        }
        EndNode(false);
    }
}
