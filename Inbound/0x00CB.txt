    f = AddByte("Function");
    
    if (f == 0) { // use scroll response
        AddLong("ScrollUid");
    } else if (f == 2) { // use scroll on item
        AddLong("EquipUid");
        DecodeItem(0);
    } else if (f == 3) { // select new item
        AddLong("EquipUid");
        DecodeItem(0);
    }
}
